 45        function [y0, y] = stepImpl(obj, u, norm)
 46            obj.msp(2:obj.nWht, :) = obj.msp(1:obj.nWht-1, :);
 47            obj.msp(1,1) = u;
 48            
 49            k = 2;
 50            for i = 2: obj.n_section
 51                sm_k = obj.nWht / k;
 52                
 53                for k0=0:k-1
 54                    if mod(k0, 2) ==0
 55                        for i0=sm_k*k0 : sm_k*(k0+1)
 56                            obj.msp(i0+1, i+1) = obj.msp(i0+1,  i-1)  - obj.msp(i0+1+ sm_k, i - 1);
 57                        end
 58                    else
 59                        for i0=sm_k*k0 : sm_k*(k0+1)
 60                            obj.msp(i0+1, i+1) = obj.msp(i0+1,  i-1)  + obj.msp(i0+1- sm_k, i - 1);
 61                        end
 62                    end
 63                end
 64                    
 65                k = k * 2;
 66            end
 67            
 68            for i=1:obj.nWht
 69                if mod(k0, 2) ==0
 70                    obj.msp(i, obj.nSectionPlus) =	obj.msp(i, obj.nSectionPlus - 1)  - obj.msp(i + 1, obj.nSectionPlus - 1);
 71                else
 72                    obj.msp(i, obj.nSectionPlus) =	obj.msp(i, obj.nSectionPlus - 1) + obj.msp(i - 1, obj.nSectionPlus - 1);
 73                end
 74            end
 75
 76            v = 0.0;
 77            for i=1:(obj.nWht/2-1)
 78                i0 = i * 2;
 79				v =v +	abs(obj.msp(i0 + 1, obj.nSectionPlus)) + abs(obj.msp(i0 + 2, obj.nSectionPlus));
 80            end
 81            y0 = v;
 82            y = u*norm;
 83        end
