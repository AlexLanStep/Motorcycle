 55        function outMoment  = stepImpl(obj, moment, npoint, err)
 56            n = max(min(npoint, 200), 6);
 57            obj.nm = n;
 58            obj.m(1, 1:obj.NCount-1) = obj.m(1,  2:obj.NCount);
 59            if err==0
 60                obj.m(1, obj.NCount) = double(moment);    
 61            else
 62                obj.m(1, obj.NCount) = double(obj.m(1, obj.NCount-1));    
 63            end
 64            
 65            kodError = length(obj.x0)
 66            
 67            
 68            if (obj.nm ~= n) | (kodError==0)
 69                obj.nm = n;
 70                obj.x0 = [0:n-1];
 71                obj.xsum = sum(obj.x0(1,1:n));
 72                obj.x2sum = sum(obj.x0.^2);
 73                obj.x_sr = obj.xsum / n;
 74                obj.x_0 = obj.x0 -obj.x_sr;
 75                dx = obj.x_0.*obj.x_0;
 76                obj.x_01 = sum(obj.x_0.^2);
 77                obj.x_001 = 1/obj.x_01;
 78                obj.xsum2 = obj.xsum * obj.xsum;
 79                obj.z0 = obj.x2sum / obj.xsum2;
 80                obj.z01 = 1 / (1 - (obj.z0 * n));
 81                obj.z002 = 1 / obj.xsum;
 82            end
 83
 84            
 85            yy = obj.m(1,  obj.NCount- obj.nm +1: obj.NCount);
 86            ysum = sum(yy);
 87            y2sum = sum(yy.^2);
 88            xy = obj.x0.*yy;
 89            xysum = sum(xy);
 90            y_sr = ysum / n;
 91            y_0 = yy-y_sr;
 92            su = sum(obj.x_0.* y_0);
 93            b = su*obj.x_001;
 94            z1 = xysum*obj.z002;
 95            betta = (z1-obj.z0*ysum)*obj.z01;
 96            alfa = (ysum - betta * n) * obj.z002;
 97            outMoment = alfa*(n-round(n/3))+betta;
 98        end
