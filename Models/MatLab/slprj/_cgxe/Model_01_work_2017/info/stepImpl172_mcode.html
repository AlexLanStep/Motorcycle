 62        function y = CalcSigm(obj, k, j)
 63            y  =  1. / (1 + ((obj.alfa+k* obj.sm_alfa) * exp(-(obj.betta +k*obj.sm_bet) * (j /10  - (obj.sd-k*obj.sdsn)))));
 64        end
 65        
 66        function obj  = Slippage(obj, vBelt,  rotv)
 67            R =  obj.Rwheel  *  obj.Transfer;
 68            
 69            if rotv <= 10
 70                x = 0;
 71            else
 72                x = 1 - (vBelt)/(rotv'*R );            
 73            end
 74            obj.dSlipp( 2:obj.lSlipp) = obj.dSlipp(1:obj.lSlipp-1);
 75            obj.dSlipp(1) = x;
 76            
 77            if obj.dSlipp(obj.lSlipp) ~= 0
 78                obj.ind=0;
 79            else
 80                 sr = sum(obj.dSlipp)/obj.lSlipp;
 81                 sd = std(obj.dSlipp);
 82                 if obj.dSlipp(1)>(sr + sd)
 83                     obj.ind = min(obj.ind+1,  6);
 84                 elseif obj.dSlipp(1)<(sr - sd)
 85                     obj.ind = max(obj.ind-1, 0);
 86                 else
 87                     obj.ind = obj.ind;
 88            end
 89            if x < obj.Kreset  % если значения x меньше 
 90                obj.ind = 0;
 91            end
 92            end
 93        end

 94        
 95        function y = stepImpl(obj, signal,  vBelt,  rotv)
 96            Slippage(obj, vBelt,  rotv)
 97            i = min(round(signal), 100);
 98            y = obj.d(obj.ind, i);
 99        end
