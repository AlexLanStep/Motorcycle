 63        function y = CalcSigm(obj, k, j)
 64            y  =  1. / (1 + ((obj.alfa+k* obj.sm_alfa) * exp(-(obj.betta +k*obj.sm_bet) * (j /10  - (obj.sd-k*obj.sdsn)))));
 65        end

 67        function [k, v]  = Slippage(obj, vBelt,  rotv)
 68            z=0;
 69            
 70            if rotv <= 10
 71                x = 0;
 72            else
 73                x = 1 - (vBelt)/(rotv' * obj.R );            
 74            end
 75            v=x;
 76            obj.dSlipp( 2:obj.lSlipp) = obj.dSlipp(1:obj.lSlipp-1);
 77            obj.dSlipp(1) = x;
 78            
 79            % I - Режим формирования массива dSlipp  
 80            if obj.dSlipp(obj.lSlipp)==0
 81                return;
 82            end
 83
 84            sr = sum(obj.dSlipp)/length(obj.dSlipp);
 85            std4 = std(obj.dSlipp)*obj.kStd;
 86            % II - Режим ind == 1
 87            if obj.ind==1
 88                if obj.dSlipp(1)>(sr + std4)   
 89                    obj.ind=min(obj.ind+1, 6);
 90                    return;
 91                end            
 92                if obj.dSlipp(1)<(sr - std4)   
 93                    obj.ind=max(obj.ind-1, 1);
 94                    return;
 95                end
 96                return;
 97            end
 98            % III - Режим ind > 1
 99            if (obj.dSlipp(1)>(sr + std4)) & (obj.dSlipp(1)>obj.dSlipp(2))
100                    obj.ind=min(obj.ind+1, 6);
101                    return;
102                end
103                if (obj.dSlipp(1)<(sr - std4)) & (obj.dSlipp(1)<obj.dSlipp(2))
104                    obj.ind=max(obj.ind-1, 1);
105                    return
106                end
107                obj.dSlipp(1)=obj.dSlipp(2);        
108        end
