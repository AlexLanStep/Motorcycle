 39        function obj = ModulSigmoid(varargin)
 40            % Support name-value pair arguments when constructing object
 41            setProperties(obj,nargin,varargin{:})
 42            obj.ind=1;
 43            obj.R =  obj.Rwheel  *  obj.Transfer;
 44            
 45             for i=1:6
 46                 for j=1:101
 47                     obj.d(i, j)= CalcSigm(obj, i-1,   j-1);       
 48                 end
 49             end            
 50             
 51             [i0, j0]  = size(obj.d);
 52             for k=i0-1:-1:1
 53%                 disp(k)
 54                 count = length(obj.d(k, obj.d(k,:)  >  obj.d(k+1, :)));
 55                 obj.d(k,1:count) = obj.d(k+1,1:count);
 56             end
 57

 66        function y = CalcSigm(obj, k, j)
 67            y  =  1. / (1 + ((obj.alfa+k* obj.sm_alfa) * exp(-(obj.betta +k*obj.sm_bet) * (j /10  - (obj.sd-k*obj.sdsn)))));
 68        end

 70        function [ k, v]  = Slippage(obj, vBelt,  rotv)
 71            k=1;
 72            x = 0;
 73            if rotv >= 10
 74                x = 1 - (vBelt)/(rotv' * obj.R );            
 75            end
 76            v=x;
 77            obj.dSlipp( 2:obj.lSlipp) = obj.dSlipp(1:obj.lSlipp-1);
 78            obj.dSlipp(1) = x;
 79            
 80            % I - Режим формирования массива dSlipp  
 81            if obj.dSlipp(obj.lSlipp)==0
 82                return;
 83            end
 84
 85            sr = sum(obj.dSlipp)/length(obj.dSlipp);
 86            std4 = std(obj.dSlipp)*obj.kStd;
 87            % II - Режим ind == 1
 88            if obj.ind==1
 89                if obj.dSlipp(1)>(sr + std4)   
 90                    obj.ind = min(obj.ind+1, 6);
 91                    k = obj.ind;
 92                    return;
 93                end            
 94                if obj.dSlipp(1)<(sr - std4)   
 95                    obj.ind=max(obj.ind-1, 1);
 96                    k = obj.ind;
 97                    return;
 98                end
 99                return;
100            end
101            % III - Режим ind > 1
102            if (obj.dSlipp(1)>(sr + std4)) & (obj.dSlipp(1)>obj.dSlipp(2))
103                    obj.ind=min(obj.ind+1, 6);
104                    k = obj.ind;
105                    return;
106                end
107                if (obj.dSlipp(1)<(sr - std4)) & (obj.dSlipp(1)<obj.dSlipp(2))
108                    obj.ind=max(obj.ind-1, 1);
109                    k = obj.ind;
110                    return
111                end
112                k = obj.ind;
113                obj.dSlipp(1)=obj.dSlipp(2);        
114        end

120        function [y, k, v, ve ]= stepImpl(obj, signal,  vBelt,  rotv)  % , ss
121            [k, v] = Slippage(obj, vBelt,  rotv);
122            y = obj.d(obj.ind, max(min(round(signal), 100), 1))
123            ye = obj.alfaExp * y +obj.alfa1 *  obj.Y0;
124            obj.Y0 = ye;
125        end
