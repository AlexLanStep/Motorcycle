 33    methods
 34        % Constructor
 35        function obj = ModulSigmoid(varargin)
 36            % Support name-value pair arguments when constructing object
 37            setProperties(obj,nargin,varargin{:})
 38            obj.ind=1;
 39
 40             for i=1:6
 41                 for j=1:101
 42                     obj.d(i, j)= CalcSigm(obj, i-1,   j-1);       
 43                 end
 44             end            
 45             
 46             [i0, j0]  = size(obj.d)
 47             for k=i0-1:-1:1
 48                 disp(k)
 49                 count = length(obj.d(k, obj.d(k,:)  >  obj.d(k+1, :)))
 50                 obj.d(k,1:count) = obj.d(k+1,1:count);
 51             end

 54

 58    methods(Access = protected)
 59        %% Common functions
 60        
 61        function y = CalcSigm(obj, k, j)
 62            y  =  1. / (1 + ((obj.alfa+k* obj.sm_alfa) * exp(-(obj.betta +k*obj.sm_bet) * (j /10  - (obj.sd-k*obj.sdsn)))));
 63        end
 64        
 65        function obj  = Slippage(obj, vBelt,  rotv)
 66            R =  obj.Rwheel  *  obj.Transfer;
 67            
 68            if rotv <= 10
 69                x = 0;
 70            else
 71                x = 1 - (vBelt)/(rotv'*R );            
 72            end
 73            
 74            obj.dSlipp( 2:obj.lSlipp) = obj.dSlipp(1:obj.lSlipp-1);
 75            obj.dSlipp(1) = x;
 76            
 77             if obj.dSlipp(obj.lSlipp) ~= 0
 78                 obj.ind=0;
 79             else
 80                 sr = sum(obj.dSlipp)/obj.lSlipp;
 81                 sd = std(obj.dSlipp);
 82                 if obj.dSlipp(1)>(sr + sd)
 83                     obj.ind = min(obj.ind+1,  6);
 84                 elseif obj.dSlipp(1)<(sr - sd)
 85                     obj.ind = max(obj.ind-1, 0);
 86                 else
 87                     obj.ind = obj.ind;
 88                 end
 89                 if x < obj.Kreset  % если значения x меньше 
 90                     obj.ind = 0;
 91                 end
 92             end
 93        end
